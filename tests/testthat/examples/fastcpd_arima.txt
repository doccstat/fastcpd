\donttest{
set.seed(1)
n <- 600
w <- rnorm(n + 1, 0, 1)
x <- rep(0, n + 1)
for (i in 1:300) {
  x[i + 1] <- 0.8 * x[i] + w[i + 1] + 0.1 * w[i]
}
for (i in 301:n) {
  x[i + 1] <- 0.1 * x[i] + w[i + 1] + 0.5 * w[i]
}
result <- fastcpd.arima(
  x[1 + seq_len(n)],
  c(1, 0, 1),
  include.mean = FALSE,
  trim = 0,
  beta = (3 + 1) * log(n) / 2 * 2
)
summary(result)
plot(result)
}
